682  function newDataOut = smoothAndInterpolate(newDataIn) {
683 
684  /* add data */
685  obj.buffer1= [obj.buffer1; newDataIn];
686  idx = (obj.p1_Npre+1):(size(obj.buffer1,1)-obj.p1_Npost);
687  nBodyContent = numel(idx);
688 
689  /* terminate if no new body data */
690  if nBodyContent == 0
691  newDataOut = [];
692  return;
693  end
694 
695  /* for (any) content data, process
696  * n.b. this could probably all be vectorised, to speed things
697  * up considerably */
698  for i = idx
699  /* fill-in missing values by linear interpolation */
700  for xy = 1:2
701  k = 1;
702  while isnan(obj.buffer1(i,xy)) && k <= obj.p1_iw
703  /* buffer1(i,xy) = (buffer1(i-1,xy) + buffer1(i+k,xy)) / 2; */
704  obj.buffer1(i,xy) = obj.buffer1(i-1,xy)*(1-1/(k+1)) + obj.buffer1(i+k,xy)*(1/(k+1));
705  k = k+1;
706  end
707  end
708  end
709 
710  /* smooth using running mean to effect a low-pass filter in
711  * the time-domain */
712  tmp = obj.buffer1;
713  m = obj.p1_ww2;
714  for i = idx
715  smoothed = mean(tmp((i-m):(i+m),1),1);
716  obj.buffer1(i,~isnan(smoothed)) = smoothed;
717  end
718 
719  /* update data store, return any items that 'overflow' the
720  * buffer */
721  newDataOut = obj.buffer1(idx,:);
722  obj.buffer1(1:nBodyContent,:) = [];
723  }
740  function newDataOut = tagEvents(newDataIn) {
741 
742  /* add data */
743  obj.buffer2= [obj.buffer2; newDataIn];
744  idx = (obj.p2_Npre+1):(size(obj.buffer2,1)-obj.p2_Npost);
745  nBodyContent = numel(idx);
746 
747  /* terminate if no new body data */
748  if nBodyContent == 0
749  newDataOut = [];
750  return;
751  end
752 
753  /* identify saccades */
754  d = [NaN; sqrt(sum(diff(obj.buffer2(:,1:2),1).^2,2))]; /* calc euchlidean distance in points */
755  d = d / obj.pixel_per_dg; /* ivis.math.IvUnitHandler.getInstance().px2deg(d);% convert to degrees visual angle */
756  dt = [NaN; diff(obj.buffer2(:,3))]; /* time diff in seconds */
757  v = d./dt; /* calc velocity (deg/sec) */
758  A = [NaN; diff(v)]; /* calc acceleration */
759  distmetric = sqrt(sum((obj.buffer2(idx-obj.p2_dOffsetPrev,1:2)-obj.buffer2(idx+obj.p2_dOffsetPost,1:2)).^2,2));
760  distmetric = distmetric / obj.pixel_per_dg; /* ivis.math.IvUnitHandler.getInstance().px2deg(distmetric);% convert to degrees visual angle */
761 
762  /* calculate the times when (any) saccades occured by looking
763  * for changes in acceleration greater than a given cutoff (default
764  * is a change in velocity > 100 degrees per second) */
765  isSaccadeOn = (A(idx) > obj.accelCriterion & v(idx) > obj.velocityCriterion & distmetric > obj.distanceCriterion);
766  isSaccadeOff = (A(idx) < -obj.accelCriterion & v(idx) > obj.velocityCriterion);
767 
768  /* identify blink as any instance with a nan in both X and Y
769  * coordinates
770  * (transient missing data points will have been interpolated above) (probably a cleaner way of doing this) */
771  isBlink = all(isnan(obj.buffer2(idx,1:2)),2);
772 
773  /* set */
774  obj.buffer2(idx,6) = 0 + isSaccadeOn*obj.SACCADE_ON_CODE + isSaccadeOff*obj.SACCADE_OFF_CODE + isBlink*obj.BLINK_CODE;
775 
776  /* % tmp?, store */
777  obj.buffer2(idx,7) = distmetric;
778  obj.buffer2(idx,8) = v(idx);
779  obj.buffer2(idx,9) = A(idx);
780 
781  /* update data store */
782  newDataOut = obj.buffer2(idx,:);
783  obj.buffer2(1:nBodyContent,:) = [];
784  }
