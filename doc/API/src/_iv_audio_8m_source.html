369  function calibrate(rms2db_fnOrMatrix,doPlot) {
370 
371  /* parse inputs */
372  if nargin < 4 || isempty(doPlot)
373  doPlot = true;
374  end
375 
376  /* load raw measurements */
377  if ischar(rms2db_fnOrMatrix)
378  if ~exist(rms2db_fnOrMatrix," file ")
379  error(" IvAudio:InvalidInput ", " Specified calibration file (%s) cannot be found ", rms2db_fnOrMatrix)
380  end
381  tmp = load(rms2db_fnOrMatrix);
382  obj.rms2db_raw= tmp.data;
383  else
384  obj.rms2db_raw= rms2db_fnOrMatrix;
385  end
386 
387  /* make initial fit */
388  rms = obj.rms2db_raw(:,1);
389  db = obj.rms2db_raw(:,2);
390  /* fit */
391  fitCoefs = polyfit(log10(rms),db,1);
392  /* compute error */
393  yhat = polyval(fitCoefs,log10(rms));
394  c = corrcoef(db, yhat);
395  r2 = c(1,end)^2; /* ALT: 1 - sum((log10(rms)-yhat).^2) / sum((log10(rms)-mean(log10(rms))).^2) */
396 
397  /* plot, if so requested */
398  if doPlot
399  figure()
400  hold on
401  plot(log10(rms),db," o ");
402  plot(log10(rms),yhat," - ");
403  end
404 
405  /* keep refitting until sufficient variance is explained, each time
406  * excluding the most outlying point (if necessary) */
407  while r2 < obj.r2_CRITERION
408  if length(rms) <= 3 /* requires 3 mininmum */
409  error(" calibration fit failed (fewer than 3 data points remaining) ");
410  end
411 
412  /* we expect that the higher values will be more trust-worthy than the
413  * lower ones, so we will weight the errors accordings */
414  weight = logspace(log10(1),log10(.1),length(db))^t;
415 
416  /* Find the biggest/most-distal outlie */
417  err = abs(db-yhat); /* work out RMS errors */
418  err = err .* weight; /* weight
419  * remove outlier */
420  rms(ismember(err,max(err))) = [];
421  db(ismember(err,max(err))) = [];
422 
423  /* refit */
424  fitCoefs = polyfit(log10(rms),db,1);
425  /* compute error */
426  yhat = polyval(fitCoefs,log10(rms));
427  c = corrcoef(db, yhat);
428  r2 = c(1,end)^2;
429  end
430 
431  /* plot, if so requested */
432  if doPlot
433  hold on
434  plot(log10(rms),db," rx ");
435  plot(log10(rms),yhat," r- ");
436  legend(" raw (initial) ", " fitted (initial) ", " raw (robust) ", " fitted (robust) ", " Location "," NorthWest ")
437  end
438 
439  /* store function handle
440  * obj.db2rms_fcn = @(rms)polyval(fitCoefs, rms); */
441  obj.db2rms_fcn= @(targLeq)exp10( (targLeq - fitCoefs(2) ) / fitCoefs(1) );
442

444  obj.useCalibration= true;

554  n = size(x,2);

556  M = Mh * 2; /* number of ramp samples */
557

732  function x_padded = padChannels(x, testChans, outChans)
733  /* Expand audio vector/matrix so that rows match N out channels
734  * (filling extra channels with zeros).
735  *
736  * @param x sound row-vector/matrix, pre padding
737  * @param testChans vector of channel indices containing non-zeros (should match the number of rows in x)
738  * @param outChans vector of (total) channel indices (e.g., [0 1])
739  * @return x_padded sound row-vector/matrix, post padding
740  *
741  * @date 26/06/14
742  * @author PRJ
743  * */
744 
745  /* init */
746  nDataChans = size(x, 1);
747  nTestChans = length(testChans);
748  nOutChans = length(outChans);
749  if nDataChans == 1 && nTestChans > 1 /* assume want to duplicate */
750  x = repmat(x, nTestChans, 1);
751  end
752  nDataChans = size(x, 1);
753  if nDataChans ~= nTestChans
754  error(" padChannels:invalidInput "," number of test channels (%i) does "" t match number of data channels (%i) ",nTestChans,nDataChans);
755  end
756  if ~all(ismember(testChans,outChans)) /* if not all of the test channels are in the output channel list */
757  testChans = testChans(:) /* #ok */
758  outChans = outChans(:) /* #ok */
759  error(" padChannels:invalidInput "," not all test channels contained within list of output channels ");
760  end
761 
762  /* do padding */
763  x_padded = zeros(nOutChans, size(x, 2));
764  for i = 1:nDataChans
765  idx = testChans(i)+1; /* +1 since matlab indexes from 1, not 0 */
766  x_padded(idx,:) = x(i,:);
767  end
768  }
786  static function x = setRMS(x,rms) {
787 
788  /* init */
789  nChans = size(x,1); /* n.b., each ROW should be an output channel */
790 
791  /* if single value given for RMS then replace with vector (identical value for each channel) */
792  if length(rms) == 1
793  rms = repmat(rms,nChans,1);
794  end
795 
796  /* similarly the sound input (x) may be a single vector, but multiple
797  * RMS values specified. In this case we'll assume that the user is
798  * being lazy and wants us to split the sound into a matrix (one row
799  * vector per channel) */
800  if size(x,1) == 1 && length(rms) > 1
801  x = repmat(x,length(rms),1);
802  nChans = size(x,1); /* recalc number of channels */
803  end
804 
805  /* ensure that RMS is a column vector (not row) */
806  if size(rms,1) == 1 /* (might just be a single item of course, in which case this will do nothing) */
807  rms = rms^t;
808  end
809 
810  /* check that each channels has an RMS value */
811  if length(rms) ~= nChans
812  error(" IvAudio:InvalidInput "," Each channel must have an RMS value (num rms = %i; num channels = %i). Input a single number to use the same value for each channel. ", length(rms), nChans);
813  end
814 
815  /* set rms */
816  for i=1:nChans
817  old_rms = sqrt(mean(x(i,:).^2));
818  x(i,:) = x(i,:).*rms(i)/old_rms;
819  end
